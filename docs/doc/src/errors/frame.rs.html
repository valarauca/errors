use std::fmt;
use std::sync::Arc;

use super::backtrace::{resolve, trace, Symbol};
use super::serde::ser::SerializeStruct;
use super::serde::{Serialize, Serializer};

#[derive(Clone)]
pub struct Frames {
    data: Arc<[Frame]>,
}
impl Serialize for Frames {
    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
        let mut s = serializer.serialize_struct("Frames", self.data.as_ref().len())?;
        for (index, frame) in self.data.as_ref().iter().enumerate() {
            s.serialize_field(NUMBA[index], frame)?;
        }
        s.end()
    }
}
impl fmt::Display for Frames {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        <Frames as fmt::Debug>::fmt(self, f)
    }
}
impl fmt::Debug for Frames {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let mut info = f.debug_struct("Stack");
        for (index, frame) in self.data.as_ref().iter().enumerate() {
            info.field(NUMBA[index], frame);
        }
        info.finish()
    }
}

/// creates a backtrace with well described information.
#[inline(never)]
pub fn build_backtrace(skip: usize, frames: usize) -> Frames {
    let pre_allocate = if frames > 10 { 10 } else { frames };
    let mut collection: Vec<Frame> = Vec::with_capacity(pre_allocate);
    let mut skipped_count = 0;
    let mut collected_count = 0;
    trace(|frame| -> bool {
        if skipped_count < skip {
            skipped_count += 1;
            return true;
        }
        resolve(frame.ip(), |symbol| {
            collection.push(Frame::from(symbol));
        });
        collected_count += 1;
        collected_count >= frames
    });
    Frames {
        data: Arc::from(collection.into_boxed_slice()),
    }
}

/// Frame contains the information concerning a stack frame.
pub struct Frame {
    pub name: Option<String>,
    pub address: Option<usize>,
    pub filename: Option<String>,
    pub line_number: Option<u32>,
}
impl<'a> From<&'a Symbol> for Frame {
    fn from(symbol: &'a Symbol) -> Self {
        let name = symbol.name().map(|name| format!("{}", name));
        let address = symbol
            .addr()
            .map(|addr| unsafe { ::std::mem::transmute::<*mut ::std::ffi::c_void, usize>(addr) });
        let filename = symbol
            .filename()
            .map(|path| path.to_string_lossy().to_string());
        let line_number = symbol.lineno();
        Frame {
            name,
            address,
            filename,
            line_number,
        }
    }
}
impl Frame {
    fn count_fields(&self) -> usize {
        let mut output = 0usize;
        if self.name.is_some() {
            output += 1;
        }
        if self.address.is_some() {
            output += 1;
        }
        if self.filename.is_some() {
            output += 1;
        }
        if self.line_number.is_some() {
            output += 1;
        }
        output
    }
}
impl fmt::Display for Frame {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        <Frame as fmt::Debug>::fmt(self, f)
    }
}
impl fmt::Debug for Frame {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let mut s = f.debug_struct("BackTraceFrame");
        if let Option::Some(ref name) = &self.name {
            s.field("name", name);
        }
        if let Option::Some(ref filename) = &self.filename {
            s.field("filename", filename);
        }
        if let Option::Some(ref line_number) = &self.line_number {
            s.field("line_number", line_number);
        }
        if let Option::Some(ref address) = &self.address {
            s.field("line_number", &format!("{:#016X}", address));
        }
        s.finish()
    }
}
impl Serialize for Frame {
    fn serialize<S: Serializer>(&self, serializer: S) -> Result<S::Ok, S::Error> {
        let mut s = serializer.serialize_struct("BackTraceFrame", self.count_fields())?;
        if let Option::Some(ref name) = &self.name {
            s.serialize_field("name", name)?;
        }
        if let Option::Some(ref filename) = &self.filename {
            s.serialize_field("filename", filename)?;
        }
        if let Option::Some(ref line_number) = &self.line_number {
            s.serialize_field("line_number", line_number)?;
        }
        if let Option::Some(ref address) = &self.address {
            s.serialize_field("line_number", &format!("{:#016X}", address))?;
        }
        s.end()
    }
}
